#include <iostream>
#include<string>
#include<climits>
#include<vector>
using namespace std;

// 1. Memorization
bool isPalindrome(string str,int st,int end){
    while(st<end){
        if(str[st]!=str[end]){
            return false;
        }
        st++;
        end--;
    }
    return true;
}

int minPalCut(string s, int st,int end,int dp[][100]){
    if(isPalindrome(s,st,end)){
        return 0;
    }
    if(dp[st][end]!=0){
        return dp[st][end];
    }
    int ans=INT_MAX;
    for(int cp=st;cp<end;cp++){
        int x=minPalCut(s,st,cp,dp);
        int y=minPalCut(s,cp+1,end,dp);
        int total=x+y+1;

        if(total<ans){
            ans=total;
        }
    }
    return dp[st][end]=ans;
}

// 2. Tabulated
int minPalCut2(string str){
    //first matrix
    vector<vector<bool> > dp(str.length(),vector<bool>(str.length()));
    for(int d=0;d<str.length();d++){
        int s=0;
        int e=d;
        while(e<str.length()){
            if(d==0){
                dp[s][e]=1;
            }
            else if(d==1){
                if(str[s]==str[e]){
                    dp[s][e]=1;
                }
            }
            else{
                if(str[s]==str[e] && dp[s+1][e-1]==1){
                    dp[s][e]=1;
                }
            }
            s++;
            e++;
        }
    }
    // second matrix
    vector<vector<int> > dp2(str.length(),vector<int>(str.length()));
    for(int d=0;d<str.length();d++){
        int s=0;
        int e=d;
        //if dp is T -> dp2 is false;
        //for d=1, if dp is F-> dp2 is T
        //matrixmultiplication Like
        while(e<str.length()){
            if(d==0 || dp[s][e]==1){
                dp2[s][e]=0;
            }
            else if(d==1){
                if(dp[s][e]==0){
                    dp2[s][e]=1;
                }
            }
            else{
                int minVal=INT_MAX;
                for(int cp=s;cp<e;cp++){
                    int fp=dp2[s][cp];
                    int sp=dp2[cp+1][e];
                    int x=fp+sp+1;
                    if(x<minVal){
                        minVal=x;
                    }
                }
                dp2[s][e]=minVal;
            }
            s++;
            e++;
        }
    }
    return dp2[0][str.length()-1];
}

    string s="abccbc";
    int dp[100][100]={0};
    cout<<minPalCut(s,0,s.length()-1,dp)<<endl;

    cout<<minPalCut2(s);
}
